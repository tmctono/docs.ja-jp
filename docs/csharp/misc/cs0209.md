 <span data-ttu-id="19f72-107">次の例では CS0209 が生成されます。</span><span class="sxs-lookup"><span data-stu-id="19f72-107">The following sample generates CS0209:</span></span>  
  
```csharp  
// CS0209.cs  
// compile with: /unsafe  
  
class Point  
{  
   public int x, y;  
}  
  
public class MyClass  
{  
   unsafe public static void Main()  
   {  
      Point pt = new Point();  
  
      fixed (int i)    // CS0209  
      {  
      }  
      // try the following lines instead  
      /*  
      fixed (int* p = &pt.x)  
      {  
      }  
      fixed (int* q = &pt.y)  
      {  
      }  
      */  
   }  
}  
```
