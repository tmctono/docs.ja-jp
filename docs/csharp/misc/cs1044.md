 <span data-ttu-id="e3dd6-105">正しくないステートメントがコンパイラによって検出されました。</span><span class="sxs-lookup"><span data-stu-id="e3dd6-105">The compiler found an invalid statement.</span></span>

{  
   public void Dispose()  
   {  
      Console.WriteLine("Res1.Dispose()");

   }  
}
